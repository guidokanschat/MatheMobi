\section{Mehrdimensionale Funktionen}

\begin{frame}{Arten mehrdimensionaler Funktionen}
\begin{enumerate}
    \item Funktionen einer Variablen, die in $\R^n$ abbilden (Wege):
    \begin{gather*}
        \vf\colon \R\to \R^n,\qquad t\mapsto \vf(t)
    \end{gather*}
    \item Funktionen mehrerer Variablen, die nach $\R$ abbilden:
    \begin{gather*}
        f\colon \R^m\to \R,\qquad \vx \mapsto f(\vx)
    \end{gather*}
    \item Funktionen mehrerer Variablen, die in $\R^n$ abbilden:
    \begin{gather*}
        \vf\colon \R^m \to \R^n,\qquad \vx \mapsto \vf(\vx)
    \end{gather*}
\end{enumerate}
Notation: wir schreiben Vektorgrößen als $\vx,\vy,\vf$ zur Unterscheidung von Zahlen $t,x,f$.
\end{frame}

\subsection{Partielle Ableitungen}
\mrnote{Provisorisch sind die Folien aus Carl Herrmanns Teil zu partiellen Ableitungen eingefügt}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Partielle Ableitungen}

\begin{itemize}
    \item Eine Raumdimension
    \begin{itemize}
        \item Die Ableitung $\frac{df(t)}{dt}$ beschreibt die {\bf Änderung} von $f(t)$ bei infinitesimalen Veränderungen von $t$
    \end{itemize}
    \item {\bf Partielle Ableitung}: Änderung der Funktion $f(x_1,\ldots,x_n)$ bei infinitesimalen Veränderungen {\bf einer} Variablen, wobei die anderen {\bf konstant} bleiben!
\end{itemize}
    \begin{gather*}
        \frac{f(x_1,\ldots,x_k+h, \ldots, x_n) - f(x_1,\ldots, x_n)}{h} \xrightarrow{h \to 0} \frac{\partial f(x_1,\ldots,x_n)}{\partial x_k}  
    \end{gather*}
\end{frame}
 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Notation}

    \begin{itemize}

        \item Schreibweisen:
        $$
        \frac{\partial f}{\partial x_k}=\partial_{x_k} f =f_{x_k}=f_{k}
        $$
        \item Beispiele:
        \begin{xalignat*}3
            f(x,y) &= x^2+y^2 & \partial_x f &= 2x &\partial_y f &= 2y \\
            f(x,y) &= 3x^2y^3 & \partial_x f &= 6xy^3 & \partial_y f &= 9x^2y^2
        \end{xalignat*}
    \end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Mehrfache Ableitungen}

    \begin{itemize}
        \item Wie bei Funktionen einer Variablen können wir mehrfache Ableitungen berechnen
        \item Alle Kombinationen von Variablen sind möglich!
        \item z.B.\ bei $f(x_1,\ldots,x_k)$:
        $$
        f_{ij} = \frac{\partial^2 f}{\partial x_i \partial x_j} =  \frac{\partial}{\partial x_i}\left(\frac{\partial f}{\partial x_j}\right)
        $$
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle<presentation>{Vertauschung von Ableitungen}

    Bei Mehrfachableitungen spielt die Reihenfolge der Ableitungen (bei den Funktionen, die wir betrachten) keine Rolle! (Satz von Schwarz)
\vspace{1em}

    {\bf Beispiel:} $f(x,y,z) = e^{xy^2}\,\sin z$
    \begin{itemize}
        \item $f_x = y^2\,e^{xy^2}\,\sin z$ , $f_y = 2xy\,e^{xy^2}\,\sin z$
        \item bei doppelten Ableitungen gilt: $f_{xy}=f_{yx}$
    \end{itemize}
    \vspace{3cm}

\end{frame}

\begin{frame}
    \frametitle{Partielle Ableitungen und Tangente}

\begin{center}
    \includegraphics[width=.8\textwidth]{fig/mv10.png}
\end{center}
\begin{itemize}
    \item Gleichung der Tangentialebene am Punkte $(x_0,y_0)$:
    \begin{gather*}
    z-z_0 = \frac{\partial f}{\partial x}(x_0,y_0) (x-x_0) +\frac{\partial f}{\partial y}(x_0,y_0) (y-y_0)         
    \end{gather*}
    \item Normalenvektor zur Ebene:
    $$
    N = (\frac{\partial f}{\partial x}(x_0,y_0), \frac{\partial f}{\partial y}(x_0,y_0),-1)
    $$
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Beispiel}

    Wir berechnen die Gleichung der Tangentialebene für die Funktion $z=x^2-y-xe^y$ am Punkt $(x_0,y_0) = (1,0)$:
    \begin{eqnarray*}
    f_x = 2x-e^y &;& f_x(1,0) = 1 \\
    f_y = -1-xe^y &;& f_y(1,0) = -2
    \end{eqnarray*}
    Da $z_0=f(1,0) = 0$ gilt folgende Gleichung:
    $$
    z -z_0 = f_x(1,0)(x-1) + f_y(1,0)(y-0) \iff z = x-1 -2y 
    $$

    Normalenvektor:
    $$
    N = (1,-2,-1)
    $$
\end{frame}

\begin{frame}
    \frametitle{Gradient einer Funktion}

    \begin{itemize}
        \item Wir definieren den {\bf Gradienten} der Funktion $f$:
        $$
    \nabla f = \left(\begin{array}{c} \partial_{x_1}f \\ \vdots \\  \partial_{x_n}f \end{array}\right)
        $$
        \item $\nabla f(p)$ ist ein Vektor in $\mathbb{R}^n$ am Punkt $p$
        \item Beispiel: $f(x,y) = 3x^2y^3$:
        $$
        \nabla f = \left(\begin{array}{c} 6xy^3 \\ 9x^2y^2 \end{array}\right) ~~\rightarrow~~ \nabla f(1,1) = \left(\begin{array}{c} 6 \\ 9 \end{array}\right)
        $$
\end{itemize}
\end{frame}

\subsection{Richtungsableitungen}

\begin{frame}
    \frametitle{Veränderungen in beliebige Richtungen}

    \begin{itemize}
        \item Die partiellen Ableitungen $\partial_x f$ unf $\partial_y f$ beschreiben die Veränderungen von $z=f(x,y)$ wenn man sich in $x$-  bzw $y$-Richtung bewegt
        \item Wie verändert sich die Funktion, wenn man sich in eine beliebige Richtung bewegt?
        \item Ableitung entlang des Vektors $\vu \neq 0$:
          \begin{gather*}
            \partial_{\vu} f(\vx) = \lim_{h\to 0} \frac{f(\vx+h\vu)-f(\vx)}{h}
          \end{gather*}
    \end{itemize}
    \begin{center}
        \includegraphics[width=.3\textwidth]{img/mv17.png}
    \end{center}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Richtungsableitung}

    \begin{itemize}
        \item Schreibweise mit Gradient (Skalarprodukt!):
        $$
        \partial_{\vu} f = \nabla f \cdot \vu
        $$
        \item $\partial_{\vu} f(\vx)$ gibt an, wie stark die Steigung in $\vu$-Richtung am Punkt $\vx$ ist
        \item $\partial_{\vu} f(\vx)>0$ bzw. $\partial_{\vu} f(\vx)< 0$: es geht nach oben bzw.\ nach unten!
          \pause
        \item Diese Ableitung hängt von der Länge des Richtungsvektors ab
          \begin{itemize}
          \item Manche Autoren fordern $\left|\vu\right| = 1$
          \item Wir normieren bei Bedarf die Richtungsableitung
            \begin{gather*}
              D_{\vu} f (\vx) = \frac{\partial_{\vu} f(\vx)}{|\vu|}
            \end{gather*}
          \end{itemize}
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Richtung konstanter Werte}

    \begin{center}
        \includegraphics[width=.3\textwidth]{img/mv18.png}
    \end{center}

    \begin{itemize}
        \item In welche Richtung verändert sich die Funktion {\bf nicht}?
        $$
        \partial_{\vu} f = 0 ~\iff~ \nabla f \cdot \vu =0 ~\iff ~ \nabla f \perp \vu
        $$
      \item $\vu$ ist ein {\bf Tangentialvektor} zu den Höhenlinien!
      \item Der Gradient ist orthogonal zu den Höhenlinien
    \end{itemize}
  \end{frame}
  
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Anstiegsrichtung}

    \begin{center}
        \includegraphics[width=.3\textwidth]{img/mv19.png}
    \end{center}

    \begin{itemize}
        \item In welche Richtung ist die Veränderung {\bf am größten}?
        $$
        \max_{\vu} \partial_{\vu} f ~\iff ~ \vu \parallel\nabla f
        $$
        \item Wenn $\vu$ Einheitsvektor: $\vu=\frac{\nabla f}{|\nabla f|}$; daher
        $
        \partial_{\vu} f = \nabla f \cdot \frac{\nabla f}{|\nabla f|} = |\nabla f|
        $
        \item Der Gradient ist an jedem Punkt die Richtung des steilsten Anstiegs
    \end{itemize}
\end{frame}

\begin{frame}

        \item Gradient:

            \nabla f =
            \begin{pmatrix}
              -x\\-y
            \end{pmatrix}
          \end{gather*}
        \item Am Punkt $(x_0,y_0) = (1,2)$: $z_0 = \nicefrac92$
        $$
        \nabla f (x_0,y_0) = \left(\begin{array}{c}-1 \\ -2\end{array}\right)
        $$

        \end{itemize}
    \end{frame}

   %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

    \frametitle<presentation>{Beispiel}

    \begin{itemize}
        \item Größte Steigung am Punkt $(x_0,y_0)$:
        $$
            |\nabla f (x_0,y_0)| = \sqrt{\left(-1\right)^2+\left(-2\right)^2} = \sqrt{5}
        $$

        $$

    \end{itemize}
\end{frame}

\subsection{Ableitung entlang eines  Wegs}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Ableitung entlang eines Weges}
  \begin{itemize}
  \item Wir bewegen uns entlang eines Weges $\vx(t)$ im Raum
  \item Im Raum ist eine differenzierbare Funktion $f(\vx)$ definiert
  \item Wie ändert sich der Wert dieser Funktion abhängig von $t$?
  \item Kettenregel:
    \begin{gather*}
      \frac{d}{dt} f(\vx(t)) = \nabla f(\vx(t)) \cdot \vx'(t)
    \end{gather*}
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel}
  \begin{gather*}
    f(\vx) = \tfrac12(x^2+y^2),\qquad
    \nabla f =
    \begin{pmatrix}
      x\\y
    \end{pmatrix}
  \end{gather*}
  \begin{itemize}
  \item Kreislinie:
    \begin{gather*}
      \vx(t) =
      \begin{pmatrix}
        \cos t\\\sin t
      \end{pmatrix},
      \qquad
      \vx'(t) =
      \begin{pmatrix}
        -\sin t\\\cos t
      \end{pmatrix},
      \qquad
      \nabla f(\vx)\cdot \vx'(t) = 0
    \end{gather*}
    \pause
  \item Gerade durch null:
    \begin{gather*}
      \vx = \frac1{\sqrt2}
      \begin{pmatrix}
        t\\t
      \end{pmatrix},\qquad
      \vx'(t) = \frac1{\sqrt2}
      \begin{pmatrix}
        1\\1
      \end{pmatrix},
      \qquad
      \nabla f(\vx)\cdot \vx'(t) = t
    \end{gather*}
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Weglänge auf unebener Fläche}
  \begin{itemize}
  \item Bei der Länge eines Wanderweges haben wir nur die ebene
    Bewegung, nicht aber die Höhe berücksichtigt.
  \item Wir betrachten nun neben $x(t)$ und $y(t)$ auch $z(t)$, gegeben durch
    \begin{gather*}
      z(t) = f\bigl(x(t),y(t)\bigr)
    \end{gather*}
  \item Daraus ergibt sich ein Weg $\vg(t) = \bigl(x(t),y(t),z(t)\bigr)$
  \item Sein Geschwindigkeitsvektor ist:
    \begin{gather*}
      \vg'(t) =
      \begin{pmatrix}
        x'(t)\\y'(t)\\\nabla f\bigl(x(t),y(t)\bigr)\cdot \bigl(x'(t),y'(t)\bigr)
      \end{pmatrix}
    \end{gather*}
  \item Das ist nun die Geschwindigkeit, die wir in die Formel für die
    Weglänge einsetzen müssen
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Beispiel: Weglänge}
  \begin{gather*}
    x(t) = \cos t,\qquad y(t) = \sin t,\qquad f(x,y) = xy
  \end{gather*}
  \begin{itemize}
  \item Wir berechnen
    \begin{gather*}
      x'(t) = -\sin t,\quad y'(t) = \cos t,\quad
      \partial_x f(x,y) = y,\quad
      \partial_y f(x,y) = x
    \end{gather*}
  \item Der Geschwindigkeitsvektor ist
    \begin{gather*}
      \vg'(t) =
      \begin{pmatrix}
        -\sin t\\ \cos t\\ -\sin^2 t + \cos^2 t
      \end{pmatrix}
    \end{gather*}
  \item Das Quadrat seiner Länge ist
    \begin{align*}
      \left|\vg'(t)\right|^2
      &= \sin^2t+\cos^2t+ \bigl[\cos^2t-\sin^2t\bigr]^2\\
      &= 1+ \bigl[\cos^2t-\sin^2t\bigr]^2
    \end{align*}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle<presentation>{Beispiel: Weglänge (Forts.)}
  \begin{itemize}
  \item Wir benutzen $\cos^2t-\sin^2t = \cos 2t$ und erhalten
    \begin{gather*}
      \left|\vg'(t)\right|^2 = 1+\cos^2(2t)
    \end{gather*}
  \item Die Weglänge für eine Umdrehung $[0,2\pi]$ ist
    \begin{gather*}
      L = \int_0^{2\pi} \left|\vg'(t)\right|\,dt
      = \int_0^{2\pi} \sqrt{1+\cos^2(2t)}\,dt
    \end{gather*}
  \item Für dieses Integral gibt es keine Stammfunktion, numerische
    Quadratur ergibt
    \begin{gather*}
      L \approx 7.64
    \end{gather*}
    \item Vergleich mit dem \glqq{}ebenen\grqq{} Kreis: $L=2\pi\approx 6.28$
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel: ideales Gasgesetz}
      \begin{itemize}
      \item Für ein ideales Gas in einem geschlossenen Behälter gilt
        mit einer Konstanten $C$:
        \begin{gather*}
         p = C\frac{T}{V}
        \end{gather*}
      \item Wie ändert sich der Druck $p$ bei Veränderungen der
        anderen Variablen (Temperatur und Volumen) von einen
        Gleichgewichtszustand $(p_0,V_0,T_0)$?
        \pause
        \begin{equation}
          \frac{dp}{dt}
          = \frac{\partial p}{\partial T}(T_0,V_0)\,\frac{dT}{dt}
          +\frac{\partial p}{\partial V}(T_0,V_0)\,\frac{dV}{dt}
        \end{equation}
      \end{itemize}
      \pause Anmerkung: in der Physik benutzt man auch das Konzept des
      Differentials anstelle der Ableitung, lässt also überall den
      Nenner $dt$ weg.
\end{frame}
Dann fällt aber der Charakter der Ableitung weg, weshalb in der
Mathematikdidaktik lieber mit Ableitungen operiert wird. Ein
Differential allein ist nur in der Integration von Bedeutung und dort
haben wir gesehen, wie sich durch Substitution Differentiale
transformieren. Wenn wir hingegen hier das $1/dt$ weglassen, so ist
die Ableitung immer mitgedacht.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle<presentation>{Beispiel: Ideales Gasgesetz}
  
  \begin{itemize}
  \item es gilt:
    \begin{gather*}
      \frac{\partial p}{\partial T}(T_0,V_0) = C\frac{1}{V_0}
      \qquad\frac{\partial p}{\partial V}(T_0,V_0) = -C\frac{T_0}{V_0^2}      
    \end{gather*}
        \item daher:
        \begin{equation}
          \frac{dp}{dt}
          = C\left( \frac{1}{V_0} \frac{dT}{dt}
            -\frac{T_0}{V_0^2} \frac{dV}{dt}\right)
        \end{equation}
      \item Veränderung des Drucks:
        \begin{itemize}
        \item  {\bf positiv} mit Veränderung der
          Temperatur $T$
          \item {\bf negativ} mit Veränderung des Volumen $V$
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Explizite Abhängigkeit vom Wegparameter}
  \begin{gather*}
    f(t,\vx)
  \end{gather*}
  \begin{itemize}
  \item Diese Funktion hängt nicht nur vom Punkt im Raum ab, sondern
    auch von der \glqq{}Zeit\grqq
  \item Sei nun $\vx(t)$ ein Weg.
    \only<1>{\begin{Denkpause}
      was ist $\frac{d}{dt}f(t,\vx(t))$?
    \end{Denkpause}}
    \pause
  \item Mit der Kettenregel erhalten wir hier
    \begin{gather*}
      \frac{d}{dt}f(t,\vx(t)) = \partial_t f(t,\vx(t))
      + \nabla_{\vx} f(t,\vx(t)) \cdot \vx'(t),
    \end{gather*}
    wobei $\nabla_{\vx}$ der Gradient bezüglich $(x_1,\dots,x_n)$ ist.
  \end{itemize}
\end{frame}

Wie sind wir auf diese Formel gekommen?
Wir erweitern zunächst den Weg $\vx(t)$ um eine Komponente und erhalten
\begin{gather*}
  \tilde\vx(t) =
  \begin{pmatrix}
    t\\\vx(t)
  \end{pmatrix},\qquad
  \tilde\vx'(t) =
  \begin{pmatrix}
    1\\vx'(t)
  \end{pmatrix}
\end{gather*}

Damit können wir dann die gewöhnliche Ableitung entlang des Wegs einsetzen und erhalten
\begin{align*}
  \frac{d}{dt}f(t,\vx(t))
  &= \frac{d}{dt}f(\tilde\vx(t))\\
  &= \partial_t f + \sum_{k=1}^n \partial_{x_k} f x_k'.
\end{align*}

\begin{frame}{Totale vs. partielle Ableitung}
  \begin{itemize}
  \item Die \textbf{partielle} Ableitung
    \begin{gather*}
      \partial_t f\bigl(t,\vx(t)\bigr)
    \end{gather*}
    ist die Ableitung von $f$ nach dem ersten Argument
  \item Die \textbf{totale} Ableitung
    \begin{gather*}
      d_t f\bigl(t,\vx(t)\bigr)
    \end{gather*}
    ist Ableitung der gesamten Funktion nach der Zeit mit der Kettenregel
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel: Bewegung eines Moleküls in einem Stoffgemisch}
  \begin{itemize}
  \item 
    Orts- und zeitabhängige Konzentration:
    \begin{gather*}
      c(t,x) = c_0 e^{-kt} \sin\left(\frac{\pi x}{L}\right)
    \end{gather*}
    
  \item Ein Molekül bewege sich entlang des Weges:
    \begin{gather*}
      x(t) = vt
    \end{gather*}
    
  \item Dann ergibt sich entlang des Weges:
    \begin{gather*}
      c(t,x(t)) = c_0 e^{-kt} \sin\left(\frac{\pi vt}{L}\right)
    \end{gather*}
    
  \item Die totale Ableitung (Wegableitung) lautet:
    \begin{gather*}
      \frac{d}{dt} c(t,x(t)) 
      = \frac{\partial c}{\partial t} 
      + \frac{\partial c}{\partial x} \frac{dx}{dt}
    \end{gather*}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle<presentation>{Beispiel Forts.}
Berechnung der Terme:
\begin{gather*}
\frac{\partial c}{\partial t} 
= -k c_0 e^{-kt} \sin\left(\frac{\pi x}{L}\right)
\\
\frac{\partial c}{\partial x} 
= c_0 e^{-kt} \frac{\pi}{L} \cos\left(\frac{\pi x}{L}\right)
\end{gather*}

Mit \( \frac{dx}{dt} = v \) folgt:
\begin{gather*}
\frac{d c}{dt}
=
- k c_0 e^{-kt} \sin\left(\frac{\pi vt}{L}\right)
+
c_0 e^{-kt} \frac{\pi v}{L} \cos\left(\frac{\pi vt}{L}\right)
\end{gather*}

\end{frame}

Die biochemische Interpretation dieser Formeln könnte etwa wie folgt aussehen:
\begin{itemize}
\item Der Term \( \frac{\partial c}{\partial t} \) beschreibt die lokale zeitliche Änderung der Konzentration, z.\,B. durch enzymatischen Abbau oder Zerfall des Stoffes.
\item Der Term \( \frac{\partial c}{\partial x} \frac{dx}{dt} \) beschreibt die Änderung entlang des Weges, da sich das Molekül durch einen räumlichen Konzentrationsgradienten bewegt (z.\,B. Diffusion oder gerichteter Transport).
\item Die totale Ableitung \( \frac{dc}{dt} \) gibt somit die tatsächlich vom Molekül „erlebte“ Konzentrationsänderung entlang seiner Trajektorie an.
\end{itemize}

\subsection{Taylor-Reihen}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Lineare Näherung in höheren Dimensionen}
  \begin{itemize}
  \item In einer Dimension kennen wir die lineare Näherung
    \begin{gather*}
      f(t) \approx f(t_0) + f'(t_0) (t-t_0)
    \end{gather*}
    \only<1>{
      \begin{Denkpause}
        Warum ist das in höheren Dimensionen komplizierter? Was können wir tun?
      \end{Denkpause}}
    \pause
  \item Wir legen eine Gerade durch $\vx_0$ und $\vx$ und rechnen eindimensional
  \item Hilfsfunktion:
    \begin{gather*}
      g(t) = f\bigl(\vx_0+t(\vx-\vx_0)\bigr),
      \qquad g(0) = f(\vx_0),
      \qquad g(1) = f(\vx).
    \end{gather*}
  \item Lineare Näherung
    \begin{align*}
      g(1) & \approx g(0) + g'(0) \\
      f(\vx) &\approx f(\vx_0) + \nabla f(\vx_0)\cdot(\vx-\vx_0)
    \end{align*}
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel}
  \begin{gather*}
    f(x,y) = x^2 + y^2,\qquad x_0 = 1,\qquad y_0 = 1
  \end{gather*}
  \begin{itemize}
  \item Näherung im Punkt $x=y=1.1$
    \begin{align*}
      f(1.1,1.1)
      &\approx 2 +
        \begin{pmatrix}
          2\\2
        \end{pmatrix}
      \cdot
      \begin{pmatrix}
        .1\\.1
      \end{pmatrix}
      = 2.4\\
      f(1.1,1.1) &= 1.21+1.21 = 2.42
    \end{align*}
    \pause
  \item Näherung im Punkt $x=y=5$:
    \begin{align*}
      f(5,5)
      &\approx 2+
        \begin{pmatrix}
          2\\2
        \end{pmatrix}
      \cdot
      \begin{pmatrix}
        4\\4
      \end{pmatrix}
      = 32\\
      f(5,5) &= 25+25 = 50
    \end{align*}
  \end{itemize}
\end{frame}

\begin{frame}
    \begin{block}{Fazit}
    Wie in einer Dimension, so ist auch in höheren Dimensionen die
    lineare Näherung umso besser, je näher die Punkte bei einaner
    liegen
  \end{block}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle<presentation>{Beispiel}
    \begin{itemize}
        \item Können wir ohne Taschenrechner eine Näherung von $z=\sqrt{2.98^2+4.01^2}$ ausrechnen?
        \item Wir definieren die Funktion $z=f(x,y)= \sqrt{x^2+y^2}$
        \item Wir suchen eine Näherung der Funktion um den Punkt $(x_0,y_0)=(3,4)$:
        $$
        z \approx 5+ f_x(3,4)(2.98-3) + f_y(3,4)(4.01-4)
        $$
        \item es gilt $f_x = x/\sqrt{x^2+y^2}$ und $f_y = y/\sqrt{x^2+y^2}$
        \item daher $f_x(3,4) = 3/5$ und $f_y(3,4) = 4/5$
        $$
        z\approx 5 + \frac{3}{5}(-0.02) + \frac{4}{5}(0.01) = 5-\frac{2}{500} = 4.996
            $$
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Steuerung eines Bioreaktors}
  \begin{itemize}
  \item Kenngrößen
    \begin{itemize}
    \item \(G\): Glukosekonzentration
    \item \(O_2\): gelöste Sauerstoffkonzentration
    \item \(R(G,O_2)\): Wachstumsrate    
    \end{itemize}
  \item Stabiler Betriebszustand \((G^*, O_2^*)\)
  \item Die lineare Taylor-Näherung lautet:
    \begin{multline*}
      R(G, O_2)
      \approx
      R(G^*, O_2^*)\\
      +
      \frac{\partial R}{\partial G}(G^*, O_2^*)
      \left(G - G^*\right)
      +
      \frac{\partial R}{\partial O_2}(G^*, O_2^*)
      \left(O_2 - O_2^*\right)
    \end{multline*}
  \end{itemize}
\end{frame}

\begin{itemize}
\item Die Funktion \(R(G,O_2)\) beschreibt die Wachstumsrate von Mikroorganismen im Bioreaktor.
\item Der Arbeitspunkt \((G^*, O_2^*)\) entspricht einem gewünschten Betriebszustand.
\item Die lineare Näherung beschreibt, wie empfindlich die Wachstumsrate auf kleine Änderungen der Prozessgrößen reagiert.
\item Die partiellen Ableitungen geben an:
\begin{itemize}
\item wie stark das Wachstum auf Änderungen der Glukosekonzentration reagiert,
\item wie stark es auf Änderungen der Sauerstoffversorgung reagiert.
\end{itemize}
\item Dies entspricht einer lokalen Sensitivitätsanalyse und wird in der Praxis zur Prozessregelung verwendet.
\item Die Regelung sorgt dafür, dass \(G-G^*\) und \(O_2 - O_2^*\) klein bleiben, die lineare Näherung bleibt hinreichend genau
\end{itemize}

\begin{frame}{Taylor-Reihen in höheren Dimensionen}
  \begin{itemize}
  \item In einer Dimension
    \begin{gather*}
      g(t) = \sum_{k=0}^\infty \frac{g^{(k)}(t_0)}{k!} (t-t_0)
    \end{gather*}
  \item Anwendung auf die Hilfsfunktion $g(t) = f\bigl(\vx_0+t(\vx-\vx_0)\bigr)$
  \item $k=1$, hier in zwei Dimensionen, $\vx = (x,y)$
    \begin{gather*}
      \nabla f(\vx_0) (\vx-\vx_0) = \partial_xf(\vx_0)(x-x_0) + \partial_yf(\vx_0)(y-y_0)
    \end{gather*}
  \item<2-> $k=2$: Berechnung von $g''(0)$
    \begin{align*}
      \nabla \partial_xf(\vx_0)(\vx-\vx_0)
      &= \partial_{xx}f(\vx_0)(x-x_0) + \partial_{yx}f(\vx_0)(y-y_0)\\
      \nabla \partial_yf(\vx_0)(\vx-\vx_0)
      &= \partial_{xy}f(\vx_0)(x-x_0) + \partial_{yy}f(\vx_0)(y-y_0)
    \end{align*}
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle<presentation>{Darstellung des zweiten Terms}
  \begin{multline*}
    \partial_{xx}f(\vx_0)(x-x_0)^2+\partial_{yy}f(\vx_0)(y-y_0)^2\\
    +\partial_{yx}f(\vx_0)(x-x_0)(y-y_0)
    +\partial_{xy}f(\vx_0)(x-x_0)(y-y_0)
  \end{multline*}
  \begin{itemize}
  \item Schreibe als Matrix-Vektor-Produkt
    \begin{gather*}
      (\vx-\vx_0)^T \nabla^2f(\vx_0)(\vx-\vx_0),
      \qquad
      \nabla^2f(\vx_0) =
      \begin{pmatrix}
        \partial_{xx}f(\vx_0)&\partial_{xy}f(\vx_0)\\
        \partial_{yx}f(\vx_0)&\partial_{yy}f(\vx_0)
      \end{pmatrix}
    \end{gather*}
  \item Die \textbf{Hesse-Matrix} $\nabla^2f(\vx_0)$ ist symmetrisch
  \item<2-> Quadratische Näherung
    \begin{gather*}
      f(\vx) \approx f(\vx_0) + \nabla f(\vx_0)\cdot(\vx-\vx_0)
      +\frac12 (\vx-\vx_0)^T \nabla^2f(\vx_0)(\vx-\vx_0)
    \end{gather*}
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel}
  \begin{gather*}
    f(\vx) = \cos x\cos y\cos z,
    \qquad \vx_0 = (0,0,0)
  \end{gather*}
  \begin{itemize}
  \item Erste Ableitungen: $\nabla f(\vx_0) = (0,0,0)$
  \item Zweite Ableitungen:
    \begin{gather*}
      \nabla^2 f(\vx_0) =
      \begin{pmatrix}
        -1 & 0 & 0\\
        0 & -1 & 0\\
        0 & 0 & -1
      \end{pmatrix}
    \end{gather*}
  \item Wir nähern die Funktion linear im Punkt $\vx=(0.1,0.2,0.3)$
    \begin{gather*}
      f(\vx) \approx (1,1,1)
    \end{gather*}
  \item Quadratisch erhalten wir
    \begin{gather*}
      f(\vx) \approx 1- 0.1^2-0.2^2-0.3^2 = 0.86
    \end{gather*}
  \end{itemize}
\end{frame}

\subsection{Extremwertaufgaben}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
\begin{frame}{Erste Definitionen}
  \begin{itemize}
  \item \textbf{Umgebung} eines Punktes $\vx_0$: eine Menge, in deren Inneren $\vx_0$ liegt.
  \item Ein Punkt $\vx_0$ heißt \textbf{lokales} Minimum (Maximum), wenn für
    alle Punkte $\vx$ in einer Umgebung $U(\vx_0)$ gilt:
    \begin{gather*}
      f(\vx_0)
      \begin{cases}
        \le f(\vx)&\text{Minimum} \\
        \ge f(\vx)&\text{Maximum}
      \end{cases}
    \end{gather*}
  \item $\vx_0$ ist \textbf{kritischer Punkt} von $f$, falls $\nabla f(\vx_0) = 0$
  \end{itemize}
  \begin{block}{Notwendige Bedingung}
    Jedes lokale Minimum oder Maximum ist ein kritischer Punkt
  \end{block}
\end{frame}

\begin{frame}
  \begin{block}{Variationsprinzip}
    Der Punkt $\vx_0$ ist ein Minimum(Maximum) der Funktion $f$, genau
    dann, wenn für jeden Vektor $\vv\neq 0$ der Wert 0 ein lokales
    Minimum(Maximum) der Hilfsfunktion
    \begin{gather*}
      g(t) = f(\vx_0+t\vv)
    \end{gather*}
    ist.
  \end{block}
  \begin{itemize}
  \item Eindimensional!
  \item Notwendige Bedingung: $g'(0)=0=\nabla f(\vx_0)\cdot \vv$
  \item Hinreichend zusätzlich: $g''(0) \ge 0$ bzw. $g''(0) \le 0$
  \item Es handelt sich um ein striktes oder isoliertes Extremum, wenn
    die Ungleichungen strikt gelten
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Bedingung an die Hesse-Matrix}
  \begin{gather*}
    H = \nabla^2 f(\vx_0),\qquad
    \vv^TH\vv
    \begin{cases}
      \ge 0 & \text{Minimum}\\
      \le 0 & \text{Maximum}
    \end{cases}
  \end{gather*}
  \begin{itemize}
  \item Die Ungleichung muss für alle $\vv$ gelten.
  \end{itemize}
  \begin{block}{Definition}
    Eine Matrix $H$ heißt positiv bzw. negativ \textbf{definit}, wenn für alle Vektoren $\vv\neq 0$ gilt:
    \begin{gather*}
      \vv^T H \vv > 0
      \quad\text{bzw.}\quad
      \vv^T H \vv < 0
    \end{gather*}
    Sie heißt \textbf{semidefinit}, wenn in den Ungleichungen $\ge$ oder $\le$
    steht, also auch die Gleichheit erlaubt ist.
  \end{block}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Beispiel}
  \begin{gather*}
    f(x,y) = \frac{ax^2+by^2}2,
    \quad
    \nabla f(x,y) =
    \begin{pmatrix}
      ax\\by
    \end{pmatrix},\quad
    \nabla^2f(x,y) =
    \begin{pmatrix}
      a & 0\\ 0 & b
    \end{pmatrix}
  \end{gather*}
  \only<1>{
    \begin{Denkpause}
      Wir werden nun Extrema dieser Funktion abhängig von $a$ und $b$ suchen
    \end{Denkpause}}
  \pause
  \begin{itemize}
  \item Wir verschieben die Sonderfälle $a=0$ und $b=0$
  \item Es gilt $\nabla f(0,0) = 0$, daher ist der Nullpunkt ein kritischer Punkt
  \item Für die Hesse-Matrix $H = \nabla^2f(x,y)$ gilt
    \begin{gather*}
      \vv^T H \vv = av_1^2+bv_2^2
      \begin{cases}
        > 0&\text{falls } a,b > 0\\
        < 0&\text{falls } a,b < 0
      \end{cases}
    \end{gather*}
  \item Damit haben wir im ersten Fall ein Minimum, im zweiten ein Maximum
  \item Das Extremum ist isoliert
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle<presentation>{Beispiel: Darstellung der Funktionen}

    \includegraphics[width=.45\textwidth]{pic/parabol-up}
    \includegraphics[width=.45\textwidth]{pic/parabol-down}
  \end{center}
\end{frame}

\begin{frame}
  \frametitle<presentation>{Fortsetzung des Beispiels}
  \begin{itemize}
  \item<+-> Sei nun $a=0$, dann gilt $\nabla f(x,0) = 0$ für beliebiges $x$.
    \begin{itemize}
    \item Gilt $b>0$, so ist $H$ positiv semidefinit, denn
      \begin{gather*}
        \vv^TH\vv = bv_2^2 \ge 0
      \end{gather*}
    \item Alle diese Punkt sind also lokale Minima, aber nicht isoliert
    \item Für $b<0$ gilt dies analog und es sind Maxima
    \end{itemize}
  \item<+-> Sei $a=b=0$. Die Funktion is dann konstant und nach der
    Definition ist jeder Punkt ein Extremum
  \end{itemize}
  \begin{Denkpause}<+->
    Was passiert, wenn $a$ und $b$ unterschiedliches Vorzeichen haben?
  \end{Denkpause}
\end{frame}

  \begin{itemize}
  \item Sattelpunkte sind eine Art von kritischen Punkten, die es in
    einer Dimension nicht gibt.
  \item Die Hesse-Matrix ist \textbf{indefinit}, je nach Richtung sind die
    Linien durch den Punkt aufwärts und abwärts gekrümmt.
  \end{itemize}

\begin{frame}{Kriterien für definite Matrizen}
  \begin{itemize}
  \item Eine symmetrische Matrix ist positiv (negativ) definit, genau dann, wenn
    alle ihre Eigenwerte positiv (negativ) sind
  \item Eine semidefinite Matrix darf auch Nulleigenwerte haben
  \item Ein indefinite Matrix hat immer Eigenwerte beider Vorzeichen
  \end{itemize} 
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Gerschgorin-Kriterium}
  Eine Matrix ist positiv definit, wenn alle Diagonaleinträge positiv
  sind und in jeder Zeile gilt
  \begin{gather*}
    a_{ii} > \sum_{j\neq i} \left|a_{ij}\right|
  \end{gather*}

  Solche Matrizen heißen auch \textbf{diagonaldominant}.
\end{frame}

\begin{frame}{Beispiel}
  \begin{gather*}
    \begin{pmatrix}
      9 & -2 & 1 & 0 & 0 \\
      -2 & 10 & -2 & 1 & 0 \\
      1 & -2 & 11 & -2 & 1 \\
      0 & 1 & -2 & 10 & -2 \\
      0 & 0 & 1 & -2 & 9
    \end{pmatrix}
  \end{gather*}

  Die Matrix ist diagonaldominant und daher positiv definit.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Determinantenkriterium}
  In zwei Dimensionen kann man das Determinantenkriterium anwenden:
  \begin{enumerate}
  \item Ist die Determinante negativ, ist die Matrix indefinit
  \item Ist die Determinante positiv, haben beide Eigenwerte gleiches Vorzeichen
    \begin{itemize}
    \item Ist dann \(a_{11}>0\), so ist auch die Matrix positive definit
    \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}{Freie Energie eines Protein-Ligand-Systems}
  \begin{gather*}
E = E(r,\vartheta)
\end{gather*}
\begin{itemize}
\item \(r\): Abstand zwischen Ligand und Bindungstasche,
\item \(\vartheta\): Konformationswinkel der Bindungstasche.
\item Energieminimum: \((r^*, \vartheta^*)\)
\end{itemize}
\only<1>{
  \begin{Denkpause}
    Wie sieht die quadratische Näherung von $E$ um \((r^*, \vartheta^*)\) aus?
  \end{Denkpause}
}
\pause
\begin{itemize}
\item Kritischer Punkt: \(\nabla E(r^*,\vartheta^*) = 0\)
\item Hesse-Matrix
  \begin{gather*}
    H =
    \begin{pmatrix}
      \frac{\partial^2 E}{\partial r^2} & \frac{\partial^2 E}{\partial r \partial \vartheta} \\[1ex]
      \frac{\partial^2 E}{\partial \vartheta \partial r} & \frac{\partial^2 E}{\partial \vartheta^2}
    \end{pmatrix}_{(r^*,\vartheta^*)}
  \end{gather*}
  \only<3>{
  \begin{Denkpause}
    Was wissen wir über die Hesse-Matrix?
  \end{Denkpause}}
\only<4>{\item $H$ ist symmetrisch und positiv definit, weil es sich um ein Minimum handelt
\item Die Größe der Einträge von $H$ beschreibt die Steifheit der Bindung
}
\end{itemize}
\end{frame}

Das Minimum \( (r^*,\vartheta^*) \) beschreibt den stabilen gebundenen Zustand. Die Energie verhält sich in der Nähe dieses Zustands wie eine „Schüssel“ mit positiv definiter Hesse-Matrix.
Große Werte von \( \frac{\partial^2 E}{\partial r^2} \) bedeuten, dass der Abstand ist stark fixiert ist (steife Bindung).
Große Werte von \( \frac{\partial^2 E}{\partial \vartheta^2} \) bedeuten, dass die Konformation ist stark eingeschränkt ist.
Die gemischte Ableitung beschreibt Kopplung, z.\,B. dass sich die Bindungstasche beim Annähern des Liganden mitverformt.
Diese quadratische Näherung ist die Grundlage der harmonischen Approximation in der Molekülphysik.

  \end{block}
\end{frame}

\subsection{Vektorfelder und Koordinatensysteme}

\mrnote{vorläufig sind die Folien aus Carl Herrmanns Teil zu Koordinatensystemen eingefügt. Die Beispiele beziehen sich noch auf Kapitel 2, da der Themenkomplex ursprünglich dort angesiedelt war}

\begin{frame}{Mehrdimensionale Funktionen mehrerer Variablen}
  \begin{gather*}
    \vf\colon \R^n \to \R^m
  \end{gather*}
  \begin{itemize}
  \item $\vf$ ordnet jedem Punkt im Raum einen Vektor zu
    \begin{gather*}
      \vf(\vx) =
      \begin{pmatrix}
        f_1(x_1,\dots,x_n)\\
        \vdots\\
        f_m(x_1,\dots,x_n)
      \end{pmatrix}
    \end{gather*}
%  \item Anwendung der Prinzipien dieses Kapitels auf Funktionen, die
    selbst Vektoren sind
  \item Gradient wird zur \textbf{Jacobi-Matrix}
    \begin{gather*}
      \nabla \vf =
      \begin{pmatrix}
        \partial_1 f_1 & \cdots & \partial_n f_1\\
        \vdots && \vdots \\
        \partial_1 f_m & \cdots & \partial_n f_m        
      \end{pmatrix}
    \end{gather*}
    \begin{itemize}
    \item Argumente der Übersichtlichkeit halber weggelassen
    \item Merkregel: wende Gradient zeilenweise an
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Beispiel: Chemotaxis}
  \begin{itemize}
  \item Konzentration eines Signalstoffs (z.B. Nahrung)
    \begin{gather*}
      c(x,y) = c_0 e^{-(x^2+y^2)}
    \end{gather*}
  \item Bakterien folgen dem Gradienten
    \begin{gather*}
      \vf(x,y) = \nabla c(x,y) = 
      \begin{pmatrix}
        \frac{\partial c}{\partial x}\\
        \frac{\partial c}{\partial y}
      \end{pmatrix}
      =
      \begin{pmatrix}
        -2 c_0 xe^{-(x^2+y^2)}\\
        -2 c_0 ye^{-(x^2+y^2)}
      \end{pmatrix}
    \end{gather*}
  \item Eine Funktion $\vf\colon \R^n\to\R^n$ nennt man auch
    \textbf{Vektorfeld}
  \item In diesem Falle Gradientenfeld
  \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Koordinatensysteme}

    \begin{itemize}
        \item Es gibt unterschiedliche {\bf Koordinatensysteme}, um die Position eines Punktes im Raum zu beschreiben
        \item Beispiele:
        \begin{enumerate}
            \item Kartesische Koordinaten $(x,y,z,...)$
            \item Polarkoordinaten im 2d Raum: $(r,\phi)$
            \item Kugelkoordinaten im 3d Raum $(r,\theta,\phi)$
        \end{enumerate}
        \item Die Wahl des Koordinatensystems is beliebig, aber bestimmte Systeme eignen sich besser als andere, und führen zu einer einfachen Beschreibung.
    \end{itemize}
\end{frame}

 \begin{itemize}
 \item im 2-dimensionalen Raum kann jeder Punkt dargestellt werden durch
   \begin{itemize}
   \item Kartesische Koordinaten $x$ und $y$
   \item Polarkoordinaten: Radius $r$ und Winkel $\phi$ 
   \end{itemize}
   \begin{gather*}
     \begin{pmatrix}
       x\\y
     \end{pmatrix}
     =
     \begin{pmatrix}

     \end{pmatrix}
     \qquad r \ge 0
   \end{gather*}
   \begin{itemize}
   \item $\sin$ und $\cos$ sind $2\pi$-periodisch, daher Optionen
     \begin{itemize}
     \item $\phi\in [0,2\pi)$ ist intuitiv
     \item $\phi\in (-\pi,\pi]$ ist typisch auf Computern
     \end{itemize}
   \end{itemize}
 \end{itemize}
\end{frame}

\begin{frame}
  \frametitle<presentation>{Polarkoordinaten}
  Umwandlung von kartesischen in Polarkoordinaten:
  \begin{gather*}
    r=\sqrt{x^2+y^2},\qquad
    \tan\phi = \frac{r\sin\phi}{r\cos\phi} = \frac{y}{x}
  \end{gather*}
  \begin{itemize}
  \item Im Nullpunkt ist der Winkel unbestimmt
  \item Die Umkehrfunktion $\arctan$ ist definiert von $\mathbb{R}\rightarrow (-\frac{\pi}{2},\frac{\pi}{2})$
  \item Winkel zwischen $-\frac{\pi}{2}$ und $\frac{\pi}{2}$
  \item $\arctan$ sieht nur das Produkt der Vorzeichen von $x$ und $y$, nicht die individuellen Vorzeichen
  \end{itemize}
\end{frame}

\begin{frame}{Implementation der Umkehrabbildung}
  \begin{itemize}
  \item Auf Computern ist die Funktion \texttt{atan2(x,y)}
    implementiert, die zur Berechnung von $\phi$ benutzt werden
    kann.
  \item \texttt{atan2(x,y)} ist definiert als der eindeutige Winkel
    $\phi\in(-\pi,\pi]$, so dass
    \begin{gather*}
      x = r \cos \phi,\qquad y = r \sin\phi\qquad
      \text{falls } r = \sqrt{x^2+y^2} > 0
    \end{gather*}
  \item \texttt{atan2(0,0)} ist mathematisch undefiniert und in der
    Implementation meistens 0.
  \item Die Transformation von Polarkoordinaten auf kartesische ist damit
    \begin{gather*}
      \begin{pmatrix}
        r\\\phi
      \end{pmatrix}
      =
      \begin{pmatrix}
        \sqrt{x^2+y^2}\\\operatorname{atan2}(x,y)
      \end{pmatrix}
    \end{gather*}
  \end{itemize}
\end{frame}

\begin{frame}{Koordinatenwechsel}
  \begin{gather*}
    f(x,y) = \left(x^2+y^2\right)^{3/2},\qquad g(r,\phi) = r^3
  \end{gather*}
  \begin{itemize}
  \item Sei $\vT$ die Transformation von Polarkoordinaten auf kartesische Koordinaten
    \begin{gather*}
      \begin{pmatrix}
        x\\y
      \end{pmatrix}
      = \vT
      \begin{pmatrix}
        r\\\phi
      \end{pmatrix},
      \qquad
      \begin{pmatrix}
        r\\\phi
      \end{pmatrix}
      = \vT^{-1}
      \begin{pmatrix}
        x\\y
      \end{pmatrix}
    \end{gather*}
  \item Es gilt dann
    \begin{gather*}
      f\bigl(\vT(r,\phi)\bigr) = g(r,\phi),\qquad
      g\bigl(\vT^{-1}(x,y)\bigr) = f(x,y)
    \end{gather*}
  \item Diese Beziehung gilt allgemein beim Koordinatenwechsel
  \end{itemize}
\end{frame}

\begin{frame}{Kettenregel}
  Gesucht ist die Ableitung von $f$ nach $r$ und $\phi$
  \begin{itemize}
  \item $g(r,\phi)= f(r\cos\phi,r\sin\phi)$
  \item Jetzt können wir die Kettenregel der Wegableitung benutzen:
    \begin{align*}
      \partial_r g = d_r f
      &= \partial_x f\cos\phi+\partial_y f \sin\phi\\
      \partial_\phi g = d_\phi f
      &= \partial_x f\; (-r\sin\phi) + \partial_y f\; r \cos \phi
    \end{align*}
  \item Berechnen wir die Jacobi-Matrix von $\vT$, erhalten wir
    \begin{gather*}
      \nabla \vT(r,\phi) =
      \begin{pmatrix}
        \cos \phi & -r \sin \phi \\
        \sin \phi & r \cos \phi
      \end{pmatrix}      
    \end{gather*}
  \item Es gilt also (sogar allgemein für beliebige $\vT$)
    \begin{gather*}
      \nabla g(r,\phi)
      = \nabla f\bigl(\vT(r,\phi)\bigr) \nabla \vT(r,\phi)
    \end{gather*}
    \begin{itemize}
    \item Alle Gradienten enthalten die partiellen Ableitungen
    \item Gradienten werden im Matrixprodukt als Zeilenvektoren gelesen
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Anwendung auf das Beispiel}
  \begin{gather*}
    f(x,y) = \left(x^2+y^2\right)^{3/2},\qquad g(r,\phi) = r^3
  \end{gather*}
  \begin{itemize}
  \item Gradient von $f$
    \begin{gather*}
      \nabla f(x,y) =
      \begin{pmatrix}
        3x\sqrt{x^2+y^2}\\3y\sqrt{x^2+y^2}
      \end{pmatrix}
      =
      \begin{pmatrix}
        3 r^2 \cos \phi\\3 r^2 \sin\phi
      \end{pmatrix}
    \end{gather*}
  \item Ableitung nach $r$: $\nabla f$ mal erste Spalte von $\nabla \vT$
    \begin{gather*}
      d_r f(\vT(r,\phi)) =
      \begin{pmatrix}
        3 r^2 \cos \phi\\3 r^2 \sin\phi
      \end{pmatrix}\cdot
      \begin{pmatrix}
        \cos\phi\\\sin\phi
      \end{pmatrix} = 3r^2 = \partial_r g(r,\phi)
    \end{gather*}
    \pause
  \item Ableitung nach $\phi$: $\nabla f$ mal zweite Spalte von $\nabla \vT$
    \begin{gather*}
      d_\phi f(\vT(r,\phi)) =
      \begin{pmatrix}
        3 r^2 \cos \phi\\3 r^2 \sin\phi
      \end{pmatrix}\cdot
      \begin{pmatrix}
        -r\sin\phi\\r\cos\phi
      \end{pmatrix} = 0 = \partial_\phi g(r,\phi)
    \end{gather*}
  \end{itemize}
\end{frame}

\begin{frame}{Ableitung der Umkehrabbildung}
  \begin{itemize}
  \item Die Umkehrabbildung ist
    \begin{gather*}
      \begin{pmatrix}
        r\\\phi
      \end{pmatrix}
      \vT^{-1}
      \begin{pmatrix}
        x\\y
      \end{pmatrix}
      =
      \begin{pmatrix}
        \sqrt{x^2+y^2}\\\operatorname{atan2}(x,y)
      \end{pmatrix}
    \end{gather*}
  \item Ihre Jacobi-Matrix ist
    \begin{gather*}
      \nabla \left(\vT^{-1}\right) =
      \begin{pmatrix}
        \frac x{\sqrt{x^2+y^2}} & \frac y{\sqrt{x^2+y^2}}\\
        \frac {-y}{x^2+y^2} & \frac x{x^2+y^2}
      \end{pmatrix}
      =
      \begin{pmatrix}
        \cos\phi & \sin \phi\\
        \tfrac{-\sin\phi}r & \tfrac{\cos\phi}r
      \end{pmatrix}
    \end{gather*}
  \item Es gilt
    \begin{gather*}
      \nabla \left(\vT^{-1}\right) \nabla \vT\
      = \begin{pmatrix}
          \cos\phi & \sin \phi\\
          \tfrac{-\sin\phi}r & \tfrac{\cos\phi}r
        \end{pmatrix}
              \begin{pmatrix}
        \cos \phi & -r \sin \phi \\
        \sin \phi & r \cos \phi
      \end{pmatrix} =
      \begin{pmatrix}
        1 & 0 \\ 0 & 1
      \end{pmatrix}
    \end{gather*}
  \item Es ist also $\nabla \left(\vT^{-1}\right) =  \left(\nabla \vT\right)^{-1}$ (auch dies gilt allgemein)
  \end{itemize}
\end{frame}

\begin{Vertiefung}
  Wir zeigen, dass tatsächlich ganz all gemein für eine Funktion
  $\vT\colon \R^n \t- \R^n$ und ihre Umkehrfunktion $\vS$ gilt, dass
  \begin{gather*}
    \nabla \vS = (\nabla \vT)^{-1}.
  \end{gather*}
  Nach Definition der Umkehrfunktion gilt (wo sie zumindest lokal existiert)
  \begin{gather*}
    \vS(\vT(\vx)) = \vx.
  \end{gather*}
  Wir leiten beide Seiten nach $\vx$ ab. Links steht nach der Kettenregel
  \begin{gather*}
    D \vS(\vT(\vx)) = \nabla \vS(\vT(\vx)) \nabla \vT(\vx),
  \end{gather*}
  wobei wir mit $D$ den Vektor der totalen Ableitungen nachden
  Koeffizienten von $\vx$ bezeichnen. Rechts sieht man leicht ein, dass die Jacobi-Matrix der Abbildung
  \begin{gather*}
    \operatorname{id}\colon \vx\mapsto \vx
  \end{gather*}
  die Einheitsmatrix $I$ sein muss. Daraus folgt
  \begin{gather*}
    \nabla \vS(\vT(\vx)) \nabla \vT(\vx) = I,
  \end{gather*}
  und \(\nabla \vS(\vT(\vx))\) ist daher die eindeutig definierte Inverse von \( \nabla \vT(\vx)\)
\end{Vertiefung}

\begin{frame}{Anwendung auf das Beispiel}
  \begin{gather*}
    f(x,y) = \left(x^2+y^2\right)^{3/2},\qquad g(r,\phi) = r^3
  \end{gather*}
  \begin{itemize}
  \item Gradient von $g$
    \begin{gather*}
      \nabla g(r,\phi) =
      \begin{pmatrix}
        3r^2\\ 0
      \end{pmatrix}
      =
      \begin{pmatrix}
        3x^2+3y^2\\0
      \end{pmatrix}
    \end{gather*}
  \item Ableitung nach $x$: $\nabla g$ mal erste Spalte von $\nabla \vT^{-1}$
    \begin{gather*}
      \only<presentation>{\hspace{-3em}}
      d_x g\bigl(\vT^{-1}(x,y)\bigr) =
      \begin{pmatrix}
        3x^2+3y^2\\0
      \end{pmatrix}\cdot
      \begin{pmatrix}
        \frac x{\sqrt{x^2+y^2}}\\
        \frac {-y}{x^2+y^2}
      \end{pmatrix} = 3x\sqrt{x^2+y^2} = \partial_x f(x,y)
    \end{gather*}
    \pause
  \item Die Ableitung nach $y$ funktioniert analog
  \end{itemize}  
\end{frame}

\subsection{Mehrdimensionale Integration}

\begin{frame}{Der Satz von Fubini}
  Mehrdimensionale Integrale können Richtung für Richtung berechnet
  werden. Die Reihenfolge spielt dabei keine Rolle.

  \vspace{1ex}
  
  Beispiel: Würfel $Q=[0,1]^2$
  \begin{gather*}
    \iint_Q f(\vx)\,d\vx = \int_0^1\int_0^1 f(x,y) \,dx\,dy
     = \int_0^1\int_0^1 f(x,y) \,dy\,dx
   \end{gather*}

   Dabei wird bei der Integration $dx$ die Variable $y$ konstant angenommen und umgekehrt.

   \vspace{1ex}

   \pause
   Die Notation $\iint$ und $\iiint$ für 2- und 3-dimensionale
   Integrale ist in der Mathematik weniger gebräuchlich und wird oft
   durch das einfache Integralzeichen $\int$ ersetzt. Die Dimension
   ergibt sich dann aus dem vektorwertigen Differential $d\vx$.
\end{frame}

\begin{frame}{Beispiel}
  \begin{multline*}
    \int_0^1\int_0^1 e^{\frac y3}\sin(\pi x)\,dy\,dx
    = 3\Bigl[ e^{\frac y3}\Bigr]_0^1 \int_0^1 \sin(\pi x)\,dx \\
    = 3 \Bigl[ e^{\frac y3}\Bigr]_0^1 \tfrac{-1}\pi \Bigl[\cos(\pi x)\Bigr)_0^1
  \end{multline*}
\end{frame}
