\begin{frame}
    \frametitle{Totales Differential}
    
    \begin{itemize}
        \item Wir definieren das {\bf totale Differential} der Funktion $f(x,y)$ am Punkt $(x_0,y_0)$:
        $$
        df = \frac{\partial f}{\partial x}(x_0,y_0)\,dx +\frac{\partial f}{\partial y}(x_0,y_0)\,dy
        $$
        \item $df$ ist das {\bf totale Differential} am Punkt $(x_0,y_0)$
        \item $dx$ und $dy$ sind infinitesimale VerûÊnderungen der Werte von $x$ und $y$ um den Punkt $(x_0,y_0)$
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{Beispiel: Ideales Gasgesetz}

    \begin{itemize}
        \item Ein ideales Gas erfû¥llt die Bedingung: $p = C\frac{T}{V}$ wo $C$ eine Konstante ist, $p$ der Druck, $V$ das Volumen und $T$ die Temperatur.
        \item Wir mûÑchten VerûÊnderungen des Drucks $p$ bei VerûÊnderungen der anderen Variablen (Temperatur und Volumen) beschreiben um einen Gleichgewichtszustand $(p_0,V_0,T_0)$:
        \begin{equation}
        dp = \frac{\partial p}{\partial T}(T_0,V_0)\,dT +\frac{\partial p}{\partial V}(T_0,V_0)\,dV
        \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Beispiel: Ideales Gasgesetz}

    \begin{itemize}
        \item es gilt:
        $$
        \frac{\partial p}{\partial T}(T_0,V_0) = C\frac{1}{V_0}~~~~~\frac{\partial p}{\partial V}(T_0,V_0) = -C\frac{T_0}{V_0^2}
        $$
        \item daher:
        \begin{equation}
        dp = C\left( \frac{1}{V_0}\cdot dT -\frac{T_0}{V_0^2}\cdot dV\right)
        \end{equation}
        \item VerûÊnderung des Drucks: {\bf positiv} mit VerûÊnderung der Temperatur $T$, {\bf negativ} mit VerûÊnderung des Volumen $V$!
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Linearisierung}

    \begin{itemize}
        \item Anhand der Tangentengleichung oder des totalen Differentials kûÑnnen wir eine {\bf lineare NûÊherung} fû¥r die Funktion $f(x,y)$ am Punkt $(x_0,y_0)$ angeben:
        $$
        f(x,y) \approx f(x_0,y_0) + \frac{\partial f}{\partial x}(x_0,y_0)(x-x_0) + \frac{\partial f}{\partial y}(x_0,y_0)(y-y_0)
        $$
        \item Diese NûÊherung ist nur gû¥ltig, wenn $x$ und $y$ in der NûÊhe von $x_0$ und $y_0$ liegen!
        \item Beispiel: $f(x,y) = x^2 + y^2$; $x_0=1$, $y_0=1$; $x=1.1$, $y=1.1$
        \item NûÊherung: $f(1.1,1.1) \approx 2 + 2\cdot(0.1) + 2\cdot(0.1) = 2.4$
        \item reeller Wert: $f(1.1,1.1) = 2.42$
    \end{itemize}
\end{frame}

    \frametitle{Beispiel}

    {\bf Beispiel 2: NûÊherung einer Funktion}
    \begin{itemize}
        \item KûÑnnen wir ohne Taschenrechner eine NûÊherung von $z=\sqrt{2.98^2+4.01^2}$ ausrechnen?
        \item Wir definieren die Funktion $z=f(x,y)= \sqrt{x^2+y^2}$
        \item Wir suchen eine NûÊherung der Funktion am Punkt $(x_0,y_0,z_0)=(3,4,5)$:
        $$
        z-5 \approx f_x(3,4)(2.98-3) + f_y(3,4)(4.01-4)
        $$
        \item es gilt $f_x = x/\sqrt{x^2+y^2}$ und $f_y = y/\sqrt{x^2+y^2}$
        \item daher $f_x(3,4) = 3/5$ und $f_y(3,4) = 4/5$
        $$
        z\approx 5 + \frac{3}{5}(-0.02) + \frac{4}{5}(0.01) = 5-\frac{2}{500} = 4.996
            $$
    \end{itemize}
\end{frame}
