\begin{frame}{Definition: Weg}
  Ein Weg (path) ist eine Abbildung von einem Zeitintervall $I$ in
  einen Vektorraum, zum Beispiel $\R^n$.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
\begin{frame}{Analytische Eigenschaften von Wegen}
    \begin{itemize}
        \item Stetigkeit, Differenzierbarkeit und Integrale von Wegen sind komponentenweise definiert.
        \item Ableitungen (Geschwindigkeitsvektor, velocity)
        \begin{gather*}
            \vf'(t) = \tfrac{d}{dt}
            \begin{pmatrix}
                \tfrac{d}{dt} f_1(t)\\
                \vdots\\
                \tfrac{d}{dt} f_n(t)
            \end{pmatrix}
            = \begin{pmatrix}
                f_1'(t)\\\vdots\\f_n'(t)
            \end{pmatrix}
          \end{gather*}
        \item Geschwindigkeit, speed: $\left|\vf'\right|$ (euklidische Norm)
        \item Integrale
        \begin{gather*}
            \int \vf(t)\,dt = \vF(t),
        \qquad F_i(t) = \int f_i(t)\,dt,\quad i=1,\dots,n
        \end{gather*}
    \end{itemize}
\end{frame}

\begin{frame}{Beispiele in zwei Dimensionen}
    \begin{columns}

            \end{align*}

\begin{frame}{Definition Kurve}
  Eine Kurve (curve) ist eine eindimensionale Teilmenge des $\R^n$
  \begin{itemize}
  \item Parameterdarstellung: Sei $\vf: I\to \R^n$ ein Weg, dann ist
    \begin{gather*}
      W = \bigl\{\vf(t) \in \R^n\big|
      t\in I \bigr\}
    \end{gather*}
    eine Kurve
    \begin{itemize}
    \item Die Kurve ist also die Spur, auf der man wandert
    \item Sie ist der anschauliche Weg der täglichen Erfahrung
    \end{itemize}
    \pause
  \item Nullstellendarstellung: Sei $f:\R^n \to \R$ eine Funktion, dann
    kann eine Kurve dargestellt werden als
    \begin{gather*}
      W = \bigl\{\vx\in\R^n\big| f(\vx) \only<3->{{\color{red}-C}} = 0
      \bigr\}
    \end{gather*}
    \begin{itemize}
    \item Höhenlinien
    \end{itemize}
  \end{itemize}

    \end{center}
  }
\end{frame}

\begin{frame}{Zur Begriffsverwirrung}
  Die Verwendung der Begriffe \glqq{}Weg\grqq{} und
  \glqq{}Kurve\grqq{} ist in der Literatur sehr uneinheitlich. Hier eine
  Tabelle einiger Standardreferenzen:
  \begin{center}
    \begin{tabular}{l|l|l}
      \hline \textbf{hier:} & \textbf{Weg} & \textbf{Kurve} \\\hline
      Wikipedia & Weg & Kurve \\
      Fritsche & Weg & Spur \\
      Hildebrandt & Kurve & Weg \\
      Forster & Kurve & --- \\
      Heuser & Weg & Bogen \\
      Königsberger & Kurve & Spur \\
      \hline
    \end{tabular}
  \end{center}
  Eine sichere Bezeichnung ist \glqq{}parametrische Kurve\grqq{} für
  das, was hier als Weg bezeichnet wird.
\end{frame}

\begin{frame}{Darstellung von Kurven und Wegen}
  \begin{itemize}
  \item Kurven im $\R^2$ kann man gut als Linien in der Ebene darstellen
    \begin{itemize}
    \item Beispiele: Landkarte, Kreislinie
    \end{itemize}
  \item Wege sind dann Bewegungen entlang dieser Linien
    \begin{itemize}
    \item Filme (s. Moodle), Snapshots
    \item 3D Linien
    \end{itemize}
  \item Wege können auch als Satz von Funktionen über der Zeitachse
    dargestellt werden
  \end{itemize}
\end{frame}

\begin{frame}{Funktionsgraphen sind Kurven}
  Sei $f(t)$ eine stetige Funktion auf dem Intervall $[a,b]$. Dann ist
  \begin{gather*}
    \vf(t) =
    \begin{pmatrix}
      t \\ f(t)
    \end{pmatrix}
  \end{gather*}
  ein Weg und der Graph der Funktion
  \begin{gather*}
    G = \left\{
      \begin{pmatrix}
        t\\f(t)
      \end{pmatrix}
      \middle|
      t\in [a,b] \right\}
  \end{gather*}
  eine Kurve.
\end{frame}

\begin{frame}
    \frametitle{Darstellung eines Weges als Funktion}

    Unter welchen Bedingungen lässt sich ein Weg auch als \grqq klassischer \grqq Graph einer Funktion $y=f(x)$ betrachten?

    \begin{itemize}
        \item Bewegung eines Punktes auf einen festen Kreis
    \begin{columns}
        \begin{column}{0.5\textwidth}
    $$
    \left\{\begin{array}{l}
        x(t) = a\cos(\omega t)\\ y(t) = a\sin(\omega t)
    \end{array}\right.
    $$
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=0.5\textwidth]{fig/pk03.png}
\end{column}
\end{columns}
    \item $\sin(\omega t) = \sqrt{1-\cos^2(\omega t)}$ (nur für die obere Kreishälfte!)
    \item Daher:
    \begin{multline}
    y(t) = a\sin(\omega t) \Longrightarrow y(t) = a \sqrt{1-\cos^2(\omega t)} \\ 
    \Longrightarrow y(t) = \sqrt{a^2-a^2\cos^2(\omega t)} \Longrightarrow y(t) = \sqrt{a^2-x(t)^2} 
    \end{multline}
    \item Also: $y = f(x) = \sqrt{a^2-x^2}$ (obere Kreishälfte)
\end{itemize}


\end{frame}
