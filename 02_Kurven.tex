\begin{frame}{Definition: Weg}
  Ein Weg (path) ist eine Abbildung von einem Zeitintervall $[a,b]$ in
  einen Vektorraum, zum Beispiel $\R^n$.

  Wir schreiben
  \begin{gather*}
    \vf(t) =
    \begin{pmatrix}
      f_1(t)\\\vdots\\f_n(t)
    \end{pmatrix},
  \end{gather*}
  also eine Funktion, die jeder Zahl $t$ einen Vektor zuordnet, oder
  einen Vektor von Funktionen.

  \pause

  Auf diese Funktionen kann man die üblichen Operationen der linearen
  Algebra anwenden, zum beispiel die euklidische Länge berechnen:
  \begin{gather*}
    \left| \vf(t) \right|
      = \sqrt{f_1(t)^2 + f_2(t)^2 + \dots + f_n(t)^2}
  \end{gather*}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
\mrwarning{Im Video sollte kurz erklärt werden, was die euklidische Norm ist}
\begin{frame}{Analytische Eigenschaften von Wegen}
    \begin{itemize}
        \item Stetigkeit, Differenzierbarkeit und Integrale von Wegen sind komponentenweise definiert.
        \item Ableitungen (Geschwindigkeitsvektor, velocity)
        \begin{gather*}
            \vf'(t) = \tfrac{d}{dt}
            \begin{pmatrix}
              f_1(t)\\
              \vdots\\
              f_n(t)
            \end{pmatrix}
            =
            \begin{pmatrix}
              \tfrac{d}{dt} f_1(t)\\
              \vdots\\
              \tfrac{d}{dt} f_n(t)
            \end{pmatrix}
            = \begin{pmatrix}
                f_1'(t)\\\vdots\\f_n'(t)
            \end{pmatrix}
          \end{gather*}
        \item Geschwindigkeit, speed: $\left|\vf'\right|$ (euklidische Länge)
        \item Integrale
          \begin{gather*}
            \int \vf(t)\,dt = \vF(t),
            \qquad F_i(t) = \int f_i(t)\,dt,\quad i=1,\dots,n
          \end{gather*}
    \end{itemize}
\end{frame}

\begin{frame}{Die Weglänge}
  \only<1>{\begin{Denkpause}
    Wie kann die Länge des Weges berechnet werden, den man vom Beginn
    der Wanderung bis zum Zeitpunkt $t$ zurückgelegt hat?
  \end{Denkpause}}
  \pause
\begin{itemize}
\item Die Strecke ist das Integral über die Geschwindigkeit
\item Die Geschwindigkeit ist die euklidische Länge der Ableitung des
  Wegs
\item Damit ist die Länge des Weges $\vf$, der im Zeitintervall $[0,T]$ zurückgelegt wurde
\begin{gather*}
    L(T) = \int_0^T \left|\vf'(t)\right| \,dt.
\end{gather*}
\end{itemize}
\end{frame}

\begin{frame}{Beispiel: Kreisumfang}
    \begin{gather*}
        \vf(t) = \begin{pmatrix}
            r \cos t\\r \sin t
        \end{pmatrix},
        \qquad
        \vf'(t) = \begin{pmatrix}
            -r \sin t\\ r\cos t
        \end{pmatrix},
        \qquad
        \left|\vf'(t)\right| = r
    \end{gather*}
    \begin{itemize}
    \item Der Kreis wird auf dem Intervall $[0,2\pi]$ einmal komplett durchlaufen
        \begin{gather*}
            \int_0^{2\pi} \left|\vf'(t)\right| \,dt = \int_0^{2\pi} r\,dt = 2\pi r
        \end{gather*}
    \end{itemize}
\end{frame}

\begin{frame}{Definition Kurve}
  Eine Kurve (curve) ist eine eindimensionale Teilmenge des $\R^n$
  \begin{itemize}
  \item Parameterdarstellung: Sei $\vf: I\to \R^n$ ein Weg, dann ist
    \begin{gather*}
      W = \bigl\{\vf(t) \in \R^n\big|
      t\in I \bigr\}
    \end{gather*}
    eine Kurve
    \begin{itemize}
    \item Die Kurve ist also die Spur, auf der man wandert
    \item Sie ist der anschauliche Weg der täglichen Erfahrung
    \end{itemize}
    \pause
  \item Nullstellendarstellung: Sei $f:\R^n \to \R$ eine Funktion, dann
    kann eine Kurve dargestellt werden als
    \begin{gather*}
      W = \bigl\{\vx\in\R^n\big| f(\vx) \only<3->{{\color{red}-C}} = 0
      \bigr\}
    \end{gather*}
    \begin{itemize}
    \item Höhenlinien
    \end{itemize}
  \end{itemize}

\begin{frame}{Zur Begriffsverwirrung}
  Die Verwendung der Begriffe \glqq{}Weg\grqq{} und
  \glqq{}Kurve\grqq{} ist in der Literatur sehr uneinheitlich. Hier eine
  Tabelle einiger Standardreferenzen:
  \begin{center}
    \begin{tabular}{l|l|l}
      \hline \textbf{hier:} & \textbf{Weg} & \textbf{Kurve} \\\hline
      Wikipedia & Weg & Kurve \\
      Fritsche & Weg & Spur \\
      Hildebrandt & Kurve & Weg \\
      Forster & Kurve & --- \\
      Heuser & Weg & Bogen \\
      Königsberger & Kurve & Spur \\
      \hline
    \end{tabular}
  \end{center}
  Eine sichere Bezeichnung ist \glqq{}parametrische Kurve\grqq{} für
  das, was hier als Weg bezeichnet wird.
\end{frame}

\begin{frame}{Charakterisierung von Parametrisierungen}
\begin{itemize}
        \item In einem Punkt der Kurve $\vx$ können sich zwei Parametrisierungen nur in der Geschwindigkeit, nicht in der Richtung unterscheiden
        \begin{itemize}
            \item Ausnahme: Kreuzungspunkte o.ä.
        \end{itemize}
        \item Deswegen können sich zwei Parametrisierungen nur durch verschiedenes Durchlaufen der Zeitachse unterscheiden
\begin{gather*}
    \biggl\{\vf(t) \bigg|\; t\in [a,b] \biggr\}
    =
    \biggl\{\vf\bigl(\phi(t)\bigr) \bigg|\; t\in \Bigl[\phi^{-1}(a),\phi^{-1}(b)\Bigr] \biggr\}
\end{gather*}
\end{itemize}
\end{frame}

\begin{frame}{Länge einer Kurve}
    \only<1>{\begin{Denkpause}
        Wir haben im letzten Abschnitt die Weglänge definiert. Wie würden Sie die Länge einer Kurve definieren? Worauf müssen Sie achten?
    \end{Denkpause}}
    \pause
    \begin{itemize}
        \item Da es viele Wege gibt, die eine Kurve parametrisieren, muss die Definition der Kurvenlänge unabhängig von der Parametrisierung sein!
        \item Länge einer Parametrisierung \only<3>{{\color{red}Substitution}}
        \begin{gather*}
            L = \int_a^b \left|\vf'(t)\right|\,dt
            \only<3>{{\color{red}=\;
            \int_{\phi^{-1}(a)}^{\phi^{-1}(b)} \left|\vf'\bigl(\phi(t)\bigr)\right|
            \phi'(t)\,dt
            }}
        \end{gather*}
        \item Länge einer zweiten Parametrisierung
        \begin{gather*}
            L = \int_{\phi^{-1}(a)}^{\phi^{-1}(b)} \left|\vf'\bigl(\phi(t)\bigr)\right|
            \left|\phi'(t)\right|\,dt
        \end{gather*}
        
        \item Beide Integrale sind gleich, wenn überall $\phi'(t) \ge 0$.
    \end{itemize}
\end{frame}

\begin{frame}{Beispiel Kreisumfang}
            \begin{gather*}
        \vf(t) = \begin{pmatrix}
            r \cos t^2\\r \sin t^2
        \end{pmatrix},
        \qquad
        \vf'(t) = \begin{pmatrix}
            -2rt \sin t^2 \\ 2rt \cos t^2
        \end{pmatrix},
        \qquad
        \left|\vf'(t)\right| = 2rt
    \end{gather*}

    \begin{itemize}
    \item Der Kreis wird auf dem Intervall $[0,\sqrt{2\pi}]$ einmal komplett durchlaufen
        \begin{gather*}
            \int_0^{\sqrt{2\pi}} \left|\vf'(t)\right| \,dt = \int_0^{\sqrt{2\pi}} 2rt\,dt = \Bigl[rt^2\Bigr]_0^{\sqrt{2\pi}} = 2\pi r
        \end{gather*}
    \end{itemize}

    \end{Denkpause}
 \end{frame}

\begin{frame}{Funktionsgraphen sind Kurven}
  Sei $f(t)$ eine stetige Funktion auf dem Intervall $[a,b]$. Dann ist
  \begin{gather*}
    \vf(t) =
    \begin{pmatrix}
      t \\ f(t)
    \end{pmatrix}
  \end{gather*}
  ein Weg und der Graph der Funktion
  \begin{gather*}
    G = \left\{
      \begin{pmatrix}
        t\\f(t)
      \end{pmatrix}
      \middle|
      t\in [a,b] \right\}
  \end{gather*}
  eine Kurve.
\end{frame}

\begin{frame}{Implizite Differentiation}
    Aus der Nullstellendarstellung $x^2+y^2=r^2$ eines Kreises vom Radius $r$ lässt sich die Ableitung $y'(x)$ berechnen:
    \begin{enumerate}
        \item Schreibe $y=y(x)$, also $x^2+y(x)^2 = r^2$
        \item Leite die Nullstellengleichung auf beiden Seiten nach $x$ ab:
        \begin{gather*}
            \frac{d}{dx}\bigl(x^2+y(x)^2)\bigr)
            = 2x + 2y(x)y'(x) = 0
        \end{gather*}
        \item Löse nach $y'(x)$ auf
        \begin{gather*}
            y'(x) = - \frac{x}{y(x)}
        \end{gather*}
    \end{enumerate}
    \pause
    Diesen Prozess der impliziten Differentiation werden wir später noch im Detail kennenlernen
\end{frame}
